namespace OpenBooking\_Exceptions;
use \Exception;

/**
 * Class NullDatasException
 * @package OpenBooking\_Exceptions
 */
class NullDatasException extends Exception
{
    /**
     * NullDatasException constructor.
     * @param string $message
     * @param int $code
     * @param Exception|null $previous
     */
    public function __construct($message, $code = -2, Exception $previous = null) {
        parent::__construct($message, $code, $previous);
    }
    /**
     * Magic Method
     * @return string
     */
    public function __toString() {
        return __CLASS__ . ": [{$this->code}]: {$this->message}\n";
    }
}
